class Solution {
  public void gameOfLife(int[][] board) {
    int m = board.length;
    int n = board[0].length;

    // Print initial board
    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        System.out.print(board[i][j] + " ");
      }
      System.out.println();
    }
  }
}

class Solution {
  public List summaryRanges(int[] nums) {
    List result = new ArrayList<>();
    if (nums.length == 0) return result;

    int start = nums[0]; // Start of the range

    for (int i = 1; i < nums.length; i++) {
      // If non-consecutive, finalize the previous range
      if (nums[i] != nums[i - 1] + 1) {
        result.add(formatRange(start, nums[i - 1]));
        start = nums[i]; // Start new range
      }
    }

    // Add the last range
    result.add(formatRange(start, nums[nums.length - 1]));
    return result;
  }

  private String formatRange(int start, int end) {
    return start == end ? String.valueOf(start) : start + "->" + end;
  }
}
